CREATE TABLE std_guardian(
    gur_id INT AUTO_INCREMENT PRIMARY KEY,
    gur_name VARCHAR(100),
    gur_address VARCHAR(100)
);

CREATE TABLE students(
    std_id INT AUTO_INCREMENT PRIMARY KEY,
    std_name VARCHAR(50),
    email VARCHAR(60),
    gur_id INT,
    FOREIGN KEY (gur_id) REFERENCES std_guardian(gur_id)
);
